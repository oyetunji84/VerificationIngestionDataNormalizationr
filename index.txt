function normalizePhoto(rawPhoto, provider) {
  
  // Step 1: Detect if URL or base64
  if (isURL(rawPhoto)) {
    // Fetch image from URL
    imageBuffer = await fetchImage(rawPhoto)
    
    // Convert to base64
    base64String = bufferToBase64(imageBuffer)
    
    // Add data URI prefix
    return `data:image/jpeg;base64,${base64String}`
  }
  
  // Step 2: Handle base64 variations
  if (hasDataURIPrefix(rawPhoto)) {
    // Already formatted correctly
    return rawPhoto
  }
  
  if (hasPartialPrefix(rawPhoto)) {
    // Extract base64 portion and re-format
    cleanBase64 = extractBase64(rawPhoto)
    return `data:image/jpeg;base64,${cleanBase64}`
  }
  
  // Step 3: Raw base64 without prefix
  return `data:image/jpeg;base64,${rawPhoto}`
}

function isURL(value) {
  return value.startsWith('http://') || value.startsWith('https://')
}

async function fetchImage(url) {
  try {
    response = await httpClient.get(url, { responseType: 'arraybuffer' })
    
    // Validate image size (e.g., max 5MB)
    if (response.data.length > 5 * 1024 * 1024) {
      throw new Error('Image size exceeds 5MB limit')
    }
    
    return response.data
    
  } catch (error) {
    // Handle fetch failures
    if (error.code === 'ENOTFOUND') {
      throw new Error('Image URL not reachable')
    }
    throw error
  }
}

async normalize(rawProviderResponse)
  INPUT: 
    - rawProviderResponse: object (inconsistent format)
  
  LOGIC:
    1. Extract and transform name fields
    2. Convert date formats to DD-MM-YYYY
    3. Standardize gender (M/F/Male/Female → Male/Female)
    4. Normalize phone format
    5. Handle image field:
       a. If URL: call imageService.convertURLToBase64()
       b. If base64: call imageService.normalizeImageField()
    6. Map all fields to unified schema
    7. Handle null/missing fields appropriately
  
  ERROR HANDLING:
    - Image fetch fails → set photo: null, log error
    - Date conversion fails → log warning, use raw value
    - Missing required fields → throw ValidationError
  
  OUTPUT:
    - Unified schema object
